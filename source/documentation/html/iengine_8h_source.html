62 struct selinfo
63 {
64  int corner;
65  int cx, cxs, cy, cys;
66  ivec o, s;
67  int grid, orient;
68  selinfo() : corner(0), cx(0), cxs(0), cy(0), cys(0), o(0, 0, 0), s(0, 0, 0), grid(8), orient(0) {}
69  int size() const { return s.x*s.y*s.z; }
70  int us(int d) const { return s[d]*grid; }
71  bool operator==(const selinfo &sel) const { return o==sel.o && s==sel.s && grid==sel.grid && orient==sel.orient; }
72  bool validate()
73  {
74  extern int worldsize;
75  if(grid <= 0 || grid >= worldsize) return false;
76  if(o.x >= worldsize || o.y >= worldsize || o.z >= worldsize) return false;
77  if(o.x < 0) { s.x -= (grid - 1 - o.x)/grid; o.x = 0; }
78  if(o.y < 0) { s.y -= (grid - 1 - o.y)/grid; o.y = 0; }
79  if(o.z < 0) { s.z -= (grid - 1 - o.z)/grid; o.z = 0; }
80  s.x = clamp(s.x, 0, (worldsize - o.x)/grid);
81  s.y = clamp(s.y, 0, (worldsize - o.y)/grid);
82  s.z = clamp(s.z, 0, (worldsize - o.z)/grid);
83  return s.x > 0 && s.y > 0 && s.z > 0;
84  }
85 };
