283 static inline void calctilebounds(float sx1, float sy1, float sx2, float sy2, int &bx1, int &by1, int &bx2, int &by2)
284 {
285  int tx1 = max(int(floor(((sx1 + 1)*0.5f*vieww)/lighttilealignw)), 0),
286  ty1 = max(int(floor(((sy1 + 1)*0.5f*viewh)/lighttilealignh)), 0),
287  tx2 = min(int(ceil(((sx2 + 1)*0.5f*vieww)/lighttilealignw)), lighttilevieww),
288  ty2 = min(int(ceil(((sy2 + 1)*0.5f*viewh)/lighttilealignh)), lighttileviewh);
289  bx1 = ((tx1 + 1) * lighttilew - 1) / lighttilevieww;
290  by1 = ((ty1 + 1) * lighttileh - 1) / lighttileviewh;
291  bx2 = (tx2 * lighttilew + lighttilevieww - 1) / lighttilevieww;
292  by2 = (ty2 * lighttileh + lighttileviewh - 1) / lighttileviewh;
293 }
